```python runnable
from random import randint

number_to_guess = randint(0, 100)
n = int(input('Guess my number! '))
while n != number_to_guess:
    if n < number_to_guess:
        print('Too small!')
    else:
        print('Too big!')
    n = int(input('Try again: '))
print('Congratulation! You found it!')
```
